#include<bits/stdc++.h>
using namespace std;
long long gcd(long long x,long long y){
	if(x%y==0)return y;
	return gcd(y,x%y);
}
long long exgcd(long long a,long long b,long long &x,long long &y){
	if(b==0){
		x=1,y=0;
		return a;
	}
	long long d=exgcd(b,a%b,y,x);
	y-=a/b*x;
	return d;
}
long long phi(long long x){
	int ans=1;
	for(int i=2;i*i<=x;i++){
		if(x%i==0){
			ans*=(i-1);
			x/=i;
			while(x%i==0)x/=i,ans*=i;
		}
	}
	if(x!=1)ans*=(x-1);
	return ans;
}
long long power(long long x,long long k,long long P){
	long long ans=1;
	while(k){
		if(k&1)(ans*=x)%=P;
		(x*=x)%=P;
		k>>=1;
	}
	return ans;
}
long long solve(long long a,long long m){
	if(m==1)return 1;
	long long M=phi(m);
	long long d=gcd(M,m);
	long long t=solve(a,d);
	long long At=power(a,t,m);
	long long x,y;
	exgcd(M/d,m/d,x,y);
	long long A=((At-t%M)/d*x)%m;
	long long ans=t%M+A*M;
	ans%=m/d*M;
	ans+=m/d*M;
	ans%=m/d*M; 
	ans+=m/d*M; 
	return ans;
}
