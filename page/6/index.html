JLOI2015 骗我呢

2018-07-08
#好题 #组合数学 #计数

题目大意：

    给出n和m，求满足$a_{i,j}<a_{i-1,j+1}$且$a_{i,j}<a_{i,j+1}$且$0\leq a_{i,j}\leq m,1\leq i\leq n,1\leq j\leq m$的方案数。

题解：

    考虑每一行，$a_{i,j}<a_{i,j+1}$，总共m个数，而值只有m+1个，意味着每一行只有一个位置的差会为2，其他都为1。

    考虑对于每上面一行，$a_{i,j}<a_{i-1,j+1}$，意味着对于第i行来说，他突变的位置最前能够比第i-1行的突变位置前一格。

    得到$O(nm)$的方程，设$f_{i,j}$表示第i行的未突变位置小于等于j的方案数。$f_{i,j}=f_{i-1,j+1}+f_{i,j-1}$。j的范围为0到m。$f_{0,0}=1$(因为小于等于相当于求前缀和，所以可以只有一个点有值)。

    考虑将对于第i行，将其第二维下标向右移i行，得到方程$f_{i,j}=f_{i-1,j}+f_{i,j-1}$。而j的范围是从i到i+m。那么原题就转化成了对一个n+1行n+m+1列的网格图(这里说的是格点)，有两条线不能越过（$y=x+1,y=x-(m+2)$），从（1,1）走到（n+1,n+m+1）的方案数。

    答案等于=全集-先越过A线的-先越过B线的。

将多次触碰同一条线视为一次触碰，那么一条线触碰AB的方案要么是ABABABAB……，要么是BABABABA……。对于先触碰A的，我们考虑先删掉以A结尾的方案，加上以BA结尾的方案，减掉以ABA结尾的方案……，直到方案数为０，就退出（每次至少向前移动一次，最多移动Ｏ（ｎ）次）。那么就可以减掉所有第一次触碰禁忌线A的方案。同理可删除先触碰B的。

    对于末尾为A的方案数计算，将终点对A线对称翻转，然后直接计算从(1,1)到对称点的方案（因为将路径沿A翻转，必定过A然后到终点。）而计算BA为末尾，就将点再沿B对称过去，计算(1,1)到对称点的方案。以此类推。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 13000000
#define P 1000000007
using namespace std;
long long fac[N],ifac[N];
int n,m;
long long power(long long x,int k){
    long long ans=1;
    while(k){
        if(k&1)(ans*=x)%=P;
        (x*=x)%=P;
        k>>=1;
    }
    return ans;
}
int pre(){
    fac[0]=1;
    for(int i=1;i<=12000000;i++)fac[i]=fac[i-1]*i%P;
    ifac[12000000]=power(fac[12000000],P-2);
    for(int i=11999999;i>=0;i--)ifac[i]=ifac[i+1]*(i+1)%P;
}
long long C(long long n,long long m){
    return fac[n]*ifac[m]%P*ifac[n-m]%P;
}
long long calc(int x,int y){
    if(x<0||y<0)return 0;
    return C(x+y,x);
}
int flipA(int &x,int &y){
    swap(x,y);
    x--;y++;
}
int flipB(int &x,int &y){
    swap(x,y);
    x+=m+2;
    y-=m+2;
}
int main(){
    pre();
    scanf("%d%d",&n,&m);
    long long ans=calc(n+m+1,n);
    int x=n+m+1,y=n;
    while(x>=0&&y>=0){
        flipA(x,y);
        ans-=calc(x,y);
        flipB(x,y);
        ans+=calc(x,y);
        ans%=P;
    }
    x=n+m+1,y=n;
    while(x>=0&&y>=0){
        flipB(x,y);
        ans-=calc(x,y);
        flipA(x,y);
        ans+=calc(x,y);
        ans%=P;
    }
    ans+=P;
    ans%=P;
    printf("%lld",ans); 
}

JLOI2015 装备购买

2018-07-08
#线性基

题目大意：

   给n个装备，每个装备有m个属性，求花费最小的基底。

题解：

    排序，贪心，然后用线性基维护即可。

#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
using namespace std;
#define eps 1e-8
const int N=505;
int n,m;
long double lb[N][N];
struct node{
    long double a[N];
    int w;
}s[N];
int tot,ans;
bool cmp(node x,node y){return x.w<y.w;}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        for(int j=1,x;j<=m;j++)scanf("%llf",&s[i].a[j]);
    for(int i=1;i<=n;i++)scanf("%d",&s[i].w);
    sort(s+1,s+1+n,cmp);
    for(int x=1;x<=n;x++)
        for(int i=1;i<=m;i++)
            if(fabs(s[x].a[i])>=eps){
                if(fabs(lb[i][i])<eps){
                    for(int j=i;j<=m;j++)lb[i][j]=s[x].a[j];
                    tot++;
                    ans+=s[x].w;
                    break;
                }
                for(int j=m;j>=i;j--)s[x].a[j]-=lb[i][j]*(s[x].a[i]/lb[i][i]);
            }
    printf("%d %d\n",tot,ans);
}

JLOI2015 城池攻占

2018-07-08
#倍增

题目描述：

    小铭铭最近获得了一副新的桌游，游戏中需要用m个骑士攻占n个城池。

    这n个城池用1到n的整数表示。除1号城池外，城池i会受到另一座城池$f_i$的管辖，其中$f_i <i$。也就是说，所有城池构成了一棵有根树。这m个骑士用1到m的整数表示，其中第i个骑士的初始战斗力为$s_i$，第一个攻击的城池为$c_i$。

    每个城池有一个防御值$h_i$，如果一个骑士的战斗力大于等于城池的生命值，那么骑士就可以占领这座城池；否则占领失败，骑士将在这座城池牺牲。占领一个城池以后，骑士的战斗力将发生变化，然后继续攻击管辖这座城池的城池，直到占领1号城池，或牺牲为止。

    除1号城池外，每个城池i会给出两个战斗力变化参数$a_i$, $v_i$。若$a_i=0$，攻占城池i以后骑士战斗力会增加$v_i$；若$a_i =1$，攻占城池i以后，战斗力会乘以$v_i$。注意每个骑士是单独计算的。也就是说一个骑士攻击一座城池，不管结果如何，均不会影响其他骑士攻击这座城池的结果。

    现在的问题是，对于每个城池，输出有多少个骑士在这里牺牲；对于每个骑士，输出他攻占的城池数量。

题解：

    考虑对于一个点来说，初始值越大，最后能到达的高度也就越高。那么直接倍增维护跳2^i所需的初始值，并将函数复合即可。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 320000
using namespace std;
long long fa[21][N],f[21][N],g[21][N],l[21][N];
long long p[N],a[N],v[N]; 
long long ans[N],bns[N];
int n,m;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%lld",&p[i]);
    for(int i=2;i<=n;i++)scanf("%lld%lld%lld",&fa[0][i],&a[i],&v[i]);
    for(int i=1;i<=n;i++){
        f[0][i]=1;g[0][i]=0;
        if(a[i]==0)g[0][i]=v[i];
        else f[0][i]=v[i];
        l[0][i]=p[i];
        for(int j=1;j<=20;j++){
            fa[j][i]=fa[j-1][fa[j-1][i]];
            if(fa[j][i]==0)break;
            f[j][i]=f[j-1][fa[j-1][i]]*f[j-1][i];
            g[j][i]=g[j-1][fa[j-1][i]]+f[j-1][fa[j-1][i]]*g[j-1][i];
            l[j][i]=max(l[j-1][i],(l[j-1][fa[j-1][i]]-g[j-1][i])/f[j-1][i]+((l[j-1][fa[j-1][i]]-g[j-1][i])%f[j-1][i]!=0));
        }
    }
    for(int i=1;i<=m;i++){
        long long x,y;
        scanf("%lld%lld",&x,&y);
        int k=i;
        for(int i=20;i>=0;i--){
            if(fa[i][y]==0)continue;
            if(x>=l[i][y]){
                bns[k]+=(1<<i);
                x=x*f[i][y]+g[i][y];
                y=fa[i][y];
            }
        }
        if(x>=p[y])bns[i]++;
        else ans[y]++;
    }
    for(int i=1;i<=n;i++)printf("%d\n",ans[i]);
    for(int i=1;i<=m;i++)printf("%d\n",bns[i]);
}

JLOI2015 有意义的字符串

2018-07-05
#矩阵快速幂 #特征方程

题目大意：

    求$[(\frac{b+\sqrt{d}}{2})^n]$ mod $7528443412579576937$。

题解：

    将式子转化成$([(\frac{b+\sqrt{d}}{2})^n]+[(\frac{b-\sqrt{d}}{2})^n])-[(\frac{b-\sqrt{d}}{2})^n]$

    将左边看成特征方程的两个解，还原特征方程。

    $x^2-bx+\frac{b^2-d}{4}=0$

    将式子左边括号中转成一个数列a，得$a_n=a_{n-1}b+a_{n-2}\frac{d-b^2}{4}$

    这个数列可通过矩阵快速幂求得，而减号后的只会为0或者1（按照数据范围讨论），特判即可。

    复杂度$O(\log^2n)$：因为这个模数很恶心，所以要再花一个log时间来快速乘法，使得不会爆longlong。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define P 7528443412579576937LL
using namespace std;
struct matrix{
    long long a[3][3];
    int clear(){memset(a,0,sizeof(a));}
}E;
long long add(long long x,long long y){
    long long ans=-P+x+y;
    if(ans<0)ans+=P;
    return ans;
}
long long mul(long long x,long long y){
    long long ans=0;
    while(y){
        if(y&1)ans=add(ans,x);
        x=add(x,x);
        y>>=1;
    }
    return ans;
}
matrix operator * (matrix a,matrix b){
    matrix ans;
    ans.clear();
    for(int i=1;i<=2;i++){
        for(int j=1;j<=2;j++){
            for(int k=1;k<=2;k++){
                ans.a[i][j]=add(mul(a.a[i][k],b.a[k][j]),ans.a[i][j]);
            }
        }
    }
    return ans;
}
matrix power(matrix x,long long k){
    matrix ans;
    ans.clear();
    ans=E;
    while(k){
        if(k&1)ans=ans*x;
        x=x*x;
        k>>=1; 
    }
    return ans;
}
int main(){
    E.a[1][1]=1;
    E.a[2][2]=1;
    long long b,d,n;
    scanf("%lld%lld%lld",&b,&d,&n);
    matrix o;
    o.clear();
    o.a[1][1]=2;
    o.a[1][2]=b;
    matrix p;
    p.clear();
    p.a[2][1]=1;
    p.a[2][2]=b;
    p.a[1][2]=(d-b*b)/4;
    o=o*power(p,n);
    printf("%lld",o.a[1][1]-(b*b!=d&&n%2==0));
}

NOI2014 购票

2018-07-05
#NOI #斜率优化 #CDQ分治 #点分治

题目大意：

    有一棵树，对于每个点v有四个参数,$s_v$,$p_v$,$q_v$,$l_v$。

    $s_v$表示当前点到父亲节点的距离。如果d为每次跳跃的距离，那么每次跳跃的费用为$dp_v+q_v$。现在所有点都要去到1号点，并且他们每次往上跳不能超过距离$l_v$。对于每个点求最小花费。

    $n\leq2*10^5$,其余所有运算不超过longlong。

题解：

    链上做法：斜率优化

    设$d_i$为1号点到i号点的距离

    $f_i=min(f_j+(d_i-d_j)p_i+q_i)$

    设选k由于j且$d_k>d_j$：

    $f_k+(d_i-d_k)p_i+q_i<f_j+(d_i-d_j)p_i+q_i$

    整理得：

    $\frac{f_k-f_j}{d_k-d_j}<p_i$

    树链剖分+凸包合并$O(n\log^3 n)$：

    对于一条链来说，直接斜率优化即可，那么对于树上，我们可以按照dfs的顺序，一次加入点，并将树树剖，依次求出f值，在插入线段树的时候，只有左右儿子都是满时才合并，因为没加入的都不会被查询，所以可在$(n\log^3n)$的时间里求出答案（树剖一条链会有log段，限制$l_i$在线段树上取区间总共有log个，每个区间二分得到答案又一个log）（因为树剖常数小所以能过qwq）（然而懒惰的我写了下面的方法。）

    CDQ+树分治+凸包合并$O(n\log^2n)$：

    对于每次分治，把重心堵住，不停地堵住，然后利用当前节点到重心的链去更新重心答案，再递归计算。（看代码应该极好理解）

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 420000
#define slope(x,y) ((f[y]-f[x]+0.0)/(d[y]-d[x]))
#define cal(i,j) (0LL+f[j]+(d[i]-d[j])*p[i]+q[i])
using namespace std;
long long n,fa[N],tot,h[N],v[N],Next[N],siz[N],g[N],cnt,Q[N],o[N];
long long p[N],q[N],L[N],d[N],w[N],f[N];
struct node{long long w,id;}a[N];
bool cmp(node x,node y){return x.w>y.w;}
int add(int x,int y,long long z){
    Next[++tot]=h[x];
    h[x]=tot;
    v[tot]=y;
    w[tot]=z;
}
int dfs(int x){
    siz[x]=1;
    for(int i=h[x];i;i=Next[i]){
        d[v[i]]=d[x]+w[i];
        dfs(v[i]);
        siz[x]+=siz[v[i]];
    }
}
int get(int x,int S,int &rt){ 
    g[x]=0; siz[x]=1;
    for(int i=h[x];i;i=Next[i]){
        if(o[v[i]])continue;
        get(v[i],S,rt);
        siz[x]+=siz[v[i]];
        g[x]=max(g[x],siz[v[i]]);
    }
    g[x]=max(g[x],S-siz[x]);
    if(g[x]<g[rt]&&siz[x]>1)rt=x;
}
int dfs2(int x){
    a[++cnt].id=x;
    a[cnt].w=d[x]-L[x];
    for(int i=h[x];i;i=Next[i])if(!o[v[i]])dfs2(v[i]);
}
int solve(int x,int S){
    if(x==2){
        x++;
        x--;
    }
    if(S==1)return 0;
    int rt=0,now;
    get(x,S,rt);
    for(int i=h[rt];i;i=Next[i])o[v[i]]=1;
    solve(x,S-siz[rt]+1);
    cnt=0;
    for(int i=h[rt];i;i=Next[i])dfs2(v[i]);
    sort(a+1,a+cnt+1,cmp);
    now=rt;
    int top,l,r,mid,pos;
    top=0;
    for(int i=1;i<=cnt;i++){
        while(now!=fa[x]&&d[a[i].id]-L[a[i].id]<=d[now]){
            while(top>1&&slope(Q[top],now)>=slope(Q[top-1],Q[top]))top--;
            Q[++top]=now;now=fa[now];
        }
        if(top>0){
            l=1;r=top;pos=1;
            while(l<=r){
                mid=(l+r)>>1;
                if(mid==top){pos=top;break;}
                if(slope(Q[mid],Q[mid+1])>=p[a[i].id])l=mid+1,pos=mid+1;
                else r=mid-1;
            }
            f[a[i].id]=min(f[a[i].id],cal(a[i].id,Q[pos]));
        }
    }
    for(int i=h[rt];i;i=Next[i])solve(v[i],siz[v[i]]);
}
int main(){
    long long x;
    scanf("%lld%lld",&n,&x);
    for(int i=2;i<=n;i++){
        scanf("%lld%lld%lld%lld%lld",&fa[i],&x,&p[i],&q[i],&L[i]);
        add(fa[i],i,x);
    }
    dfs(1);
    g[0]=n+1;
    for(int i=2;i<=n;i++)f[i]=1LL<<60;
    solve(1,siz[1]);
    for(int i=2;i<=n;i++)printf("%lld\n",f[i]);
}

NOI2014 随机数生成器

2018-07-02
#NOI #卡常 #贪心 #模拟

题目大意：

    给出一个随机生成的矩阵（数字为1到n*m），求字典序最小的排序后的从$(1,1)$到$(n,m)$的数字序列。

题解：

    按照题意模拟出来之后从小到大贪心，每次加入点之后把不可能的情况排除，直接暴力排除（遇到被排除的就退），因为每个点最多只会被排除一次，所以排除的复杂度是$O(nm)$的，模拟出矩阵也是$O(nm)$的，总复杂度为$O(n*m)$。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define swap(x,y) t=x;x=y;y=t;
using namespace std;
int x0,a,b,c,d,n,m,q,tot;
int f[26000000],p[26000000],l[10200],r[10200];
static int t;
int main(){
    scanf("%d%d%d%d%d",&x0,&a,&b,&c,&d);
    scanf("%d%d%d",&n,&m,&q);
    for(int i=1;i<=n*m;i++)f[i]=i;
    for(int i=1;i<=n*m;i++){
        x0=(1LL*a*x0*x0+1LL*b*x0+c)%d;
        swap(f[i],f[x0%i+1]);
    }
    static int x,y;
    for(static int i=1;i<=q;i++){
        scanf("%d%d",&x,&y);
        swap(f[x],f[y]);
    }
    for(static int i=1;i<=n*m;i++)p[f[i]]=i;
    for(static int i=1;i<=n*m;i++){
        bool ooo=true;
        int y=(p[i]-1)%m+1;
        int x=(p[i]-y)/m+1;
        if(p[i]==0)continue;
        for(int i=x-1;i>=1;i--){
            for(int j=y+1;j<=m;j++){
                int ooo=((i-1)*m+j);
                if(p[f[ooo]]==0)break;
                p[f[ooo]]=0;
            }
        }
        for(int i=x+1;i<=n;i++){
            for(int j=y-1;j>=1;j--){
                int ooo=((i-1)*m+j);
                if(p[f[ooo]]==0)break;
                p[f[ooo]]=0;
            }
        }
        if(ooo==false)continue;
        l[++tot]=x;
        r[tot]=y;
        printf("%d ",i);
        if(tot==n+m-1)return 0; 
    }
}

NOI2014 魔法森林

2018-07-02
#NOI #SPFA

题目大意：

    一个无向图，每条边两个值a和b，求1到n路径上的$min(max(a_i)+max(b_j))$。

题解：

    按照a为第一关键字，b为第二关键字排序，按顺序加边，再松弛，因为每条边松弛一次，所以跟整张图的SPFA是一个复杂度。相当于对于每个a的可能值都做一次b的最短路，约等于优化暴力（spaly维护最小生成树的做法我不会啊qwq）

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 12000000
using namespace std;
struct node{
    int x,y,a,b;
}a[N];
int tot,Next[N],v[N],w[N],h[N],d[N]; 
int q[N],p[N],n,m;
int add(int x,int y,int z){
    tot++;
    Next[tot]=h[x];
    v[tot]=y;
    w[tot]=z;
    h[x]=tot;
}
bool cmp(node x,node y){
    if(x.a==y.a)return x.b<y.b;
    return x.a<y.a;
}
int spfa(int x){
    q[1]=x;
    if(x==1){
        x++;
        x--;
    }
    int head=0,tail=1;
    p[x]=1;
    while(head<tail){
        int now=q[++head];
        if(d[now]==0)continue;
        for(int i=h[now];i;i=Next[i]){
            if(max(d[now],w[i])<d[v[i]]||d[v[i]]==0){
                d[v[i]]=max(d[now],w[i]);
                if(p[v[i]]==0){
                    p[v[i]]=1;
                    q[++tail]=v[i];
                }
            }
        }
        p[now]=0;
    }
    p[x]=0;
}
int main(){
    d[1]=1;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++)scanf("%d%d%d%d",&a[i].x,&a[i].y,&a[i].a,&a[i].b);
    sort(a+1,a+1+m,cmp);
    int ans=1000000000,bns=1000000000;
    for(int i=1;i<=m;i++){
        add(a[i].x,a[i].y,a[i].b);
        add(a[i].y,a[i].x,a[i].b);
        spfa(a[i].x);
        spfa(a[i].y);
        if(d[n]==0)continue;
        bns=a[i].a+d[n];
        if(bns<ans)ans=bns;
    }
    spfa(1);
    if(ans==1000000000)ans=-1;
    printf("%d",ans);
}

NOI2014 动物园

2018-07-02
#倍增 #NOI #KMP #卡常

题目大意：

    对于S的每个前缀T，对于T求出有多少个长度小于|T|的前缀和后缀相同，讲每个前缀T的答案加一乘起来膜$1000000007$。

题解：

    直接求Next数组然后倍增就好了。（倍增前后坐标相反能快差不多一倍，不然卡不过去qwq）

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 1200000
#define P 1000000007
using namespace std;
char s[N];
long long f[21][N],ans;
int Next[N],q[N],g[N];
int cal(int x){
    if(f[0][x]==0)return 0;
    int o=x;
    for(int i=20;i>=0;i--)
        if(f[i][o]>x/2)o=f[i][o];
    return f[0][o];
}
int get(){
    memset(f,0,sizeof(f));
    memset(q,0,sizeof(q));
    memset(g,0,sizeof(g));
    memset(Next,0,sizeof(Next));
    int l=strlen(s+1);
    q[1]=1;
    for(int i=2;i<=l;i++){
        int j=Next[i-1];
        while(j&&s[j+1]!=s[i])j=Next[j];
        Next[i]=j+(s[j+1]==s[i]);
        q[i]=q[Next[i]]+1;
        f[0][i]=Next[i];
        for(int j=1;j<=20;j++){
            f[j][i]=f[j-1][f[j-1][i]];
            if(f[j][i]==0)break;
        }
    }
    ans=1;
    for(int i=1;i<=l;i++)g[i]=q[cal(i)];
    for(int i=1;i<=l;i++)(ans*=1LL*g[i]+1)%=P; 
    printf("%lld\n",ans);
}

NOI2014 起床困难综合征

2018-07-02
#NOI #贪心

题目大意：

    给定一系列位运算操作，在1到m里选一个数，使这个数在经历一系列操作后值最大。

题解：

    对于每一位分别求出输入时是0和1为多少，从高位往低位贪心选择较大值，如果在相同满足要求的结果下选择输入较小的。

#include<iostream>
#include<stdio.h>
#include<string.h> 
using namespace std;
long long n,m,d,ans,l;
char s[100];
int main(){
    scanf("%lld%lld",&n,&m);
    long long x=0,y=(1LL<<32)-1;
    for(int i=1;i<=n;i++){
        scanf("%s",s+1);
        scanf("%lld",&d);
        if(s[1]=='O')x=x|d,y=y|d;
        if(s[1]=='A')x=x&d,y=y&d;
        if(s[1]=='X')x=x^d,y=y^d;
    }
    for(int i=32;i>=0;i--){
        if(x&(1LL<<(i))){
            ans+=1LL<<i;
        }else{
            if(y&(1LL<<i)){
                if(l+(1LL<<i)<=m){
                    ans+=1LL<<i;
                    l+=1LL<<i;
                }
            }
        }
    }
    printf("%lld",ans);
}

NOI2007 生成树计数

2018-06-30
#NOI #状压DP #最小表示法

题目大意：

    有一条链，链上长度不超过k的点之间都会进行连边，求生成树个数。

题解：

    60%做法：$O(n^3)$,$n \leq 100$

    那么直接连边按照，用矩阵树定理高斯消元解行列式即可。

 因为$n\leq100$那么直接连边按照，用矩阵树定理$n^3$高斯消元解行列式即可。

    100%做法：$O(l^3log(n))$,$n\leq 10^{15}$,l为所有联通形式种类，当k=5时l=52。

    考虑到对于第i-k号点，如果最后会成为生成树，他必定与i-k+1到i号点中至少一个是一个联通快的。因此，我们维护$f_{i,j}$表示，在当前第i个点，前j个点的联通块是怎么样的，每次转移则是用第i号点与前面的连边情况进行转移，维护判断i-k号点在转移后是否与i-k+1到i号点中的部分点联通。搜索我们可以得到转移矩阵。

    初始条件：因为在前k个点中，所有点之间都是两两相连的，n个点的完全图的最小生成树是$n^{n-2}$，这个可用purfer序列简易证明。

    关于最小表示法：既联通情况2,2,3,1与联通情况1,1,2,3是相同的，所有本质相同的情况我们用它的最小表示来识别它。

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<map>
#define N 120
#define P 65521
using namespace std;
int cnt,k,tot,l;
int a[10],b[10],q[60];
long long n;
map<int,int>p;
struct matrix{
    long long a[N][N];
    int clear(){memset(a,0,sizeof(a));}
    int E(){for(int i=1;i<=cnt;i++)a[i][i]=1;}
}g,f;
matrix operator *(matrix a,matrix b){
    matrix ans;
    ans.clear();
    for(int i=1;i<=cnt;i++){
        for(int j=1;j<=cnt;j++){
            for(int k=1;k<=cnt;k++){
                (ans.a[i][j]+=a.a[i][k]*b.a[k][j])%=P;
            }
        }
    }
    return ans;
}
matrix power(matrix x,long long k){
    matrix ans;
    ans.clear();
    ans.E();
    while(k){
        if(k&1)ans=ans*x;
        x=x*x;
        k>>=1;
    }
    return ans;
}
int get(int x){
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    int tot=0;
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3);
    for(int i=1;i<=k;i++){
        if(b[a[i]])a[i]=b[a[i]];
        else b[a[i]]=++tot,a[i]=tot;
    }
    int y=0;
    for(int i=1;i<=k;i++)y=y<<3,y+=a[i];
    return y;
}
int powerr(int x,int k){
    int ans=1;
    if(k<=0)return ans;
    while(k){
        if(k&1)(ans*=x)%=P;
        (x*=x)%=P;
        k>>=1;
    }
    return ans;
}
int gett(int x){
    int ans=1;
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3),b[a[i]]++;
    for(int i=1;i<=k;i++)ans*=powerr(b[i],b[i]-2);
    return ans;
}

    if(x==l){

    }
}

    int o=0;
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3);

        if(a[i]==a[1])break;
        if(y&1)continue;

    }
    for(int i=0;i<k;i++){
        if((1<<i)&y){
            if(o==0)o=a[i+1];
            if(b[a[i+1]]>0)return 0;
            b[a[i+1]]=o;
        }
    }
    for(int i=1;i<=k;i++)if(b[a[i]]>0)a[i]=o;
    int l=0;
    for(int i=2;i<=k;i++)l=(l<<3)+a[i];
    l<<=3;
    if(y!=0)l+=o;
    else l+=k+1;

}

}
