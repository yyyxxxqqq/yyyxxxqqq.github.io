JLOI2015 城池攻占

2018-07-08
#倍增

题目描述：

    小铭铭最近获得了一副新的桌游，游戏中需要用m个骑士攻占n个城池。

    这n个城池用1到n的整数表示。除1号城池外，城池i会受到另一座城池$f_i$的管辖，其中$f_i <i$。也就是说，所有城池构成了一棵有根树。这m个骑士用1到m的整数表示，其中第i个骑士的初始战斗力为$s_i$，第一个攻击的城池为$c_i$。

    每个城池有一个防御值$h_i$，如果一个骑士的战斗力大于等于城池的生命值，那么骑士就可以占领这座城池；否则占领失败，骑士将在这座城池牺牲。占领一个城池以后，骑士的战斗力将发生变化，然后继续攻击管辖这座城池的城池，直到占领1号城池，或牺牲为止。

    除1号城池外，每个城池i会给出两个战斗力变化参数$a_i$, $v_i$。若$a_i=0$，攻占城池i以后骑士战斗力会增加$v_i$；若$a_i =1$，攻占城池i以后，战斗力会乘以$v_i$。注意每个骑士是单独计算的。也就是说一个骑士攻击一座城池，不管结果如何，均不会影响其他骑士攻击这座城池的结果。

    现在的问题是，对于每个城池，输出有多少个骑士在这里牺牲；对于每个骑士，输出他攻占的城池数量。

题解：

    考虑对于一个点来说，初始值越大，最后能到达的高度也就越高。那么直接倍增维护跳2^i所需的初始值，并将函数复合即可。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 320000
using namespace std;
long long fa[21][N],f[21][N],g[21][N],l[21][N];
long long p[N],a[N],v[N]; 
long long ans[N],bns[N];
int n,m;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%lld",&p[i]);
    for(int i=2;i<=n;i++)scanf("%lld%lld%lld",&fa[0][i],&a[i],&v[i]);
    for(int i=1;i<=n;i++){
        f[0][i]=1;g[0][i]=0;
        if(a[i]==0)g[0][i]=v[i];
        else f[0][i]=v[i];
        l[0][i]=p[i];
        for(int j=1;j<=20;j++){
            fa[j][i]=fa[j-1][fa[j-1][i]];
            if(fa[j][i]==0)break;
            f[j][i]=f[j-1][fa[j-1][i]]*f[j-1][i];
            g[j][i]=g[j-1][fa[j-1][i]]+f[j-1][fa[j-1][i]]*g[j-1][i];
            l[j][i]=max(l[j-1][i],(l[j-1][fa[j-1][i]]-g[j-1][i])/f[j-1][i]+((l[j-1][fa[j-1][i]]-g[j-1][i])%f[j-1][i]!=0));
        }
    }
    for(int i=1;i<=m;i++){
        long long x,y;
        scanf("%lld%lld",&x,&y);
        int k=i;
        for(int i=20;i>=0;i--){
            if(fa[i][y]==0)continue;
            if(x>=l[i][y]){
                bns[k]+=(1<<i);
                x=x*f[i][y]+g[i][y];
                y=fa[i][y];
            }
        }
        if(x>=p[y])bns[i]++;
        else ans[y]++;
    }
    for(int i=1;i<=n;i++)printf("%d\n",ans[i]);
    for(int i=1;i<=m;i++)printf("%d\n",bns[i]);
}

JLOI2015 有意义的字符串

2018-07-05
#矩阵快速幂 #特征方程

题目大意：

    求$[(\frac{b+\sqrt{d}}{2})^n]$ mod $7528443412579576937$。

题解：

    将式子转化成$([(\frac{b+\sqrt{d}}{2})^n]+[(\frac{b-\sqrt{d}}{2})^n])-[(\frac{b-\sqrt{d}}{2})^n]$

    将左边看成特征方程的两个解，还原特征方程。

    $x^2-bx+\frac{b^2-d}{4}=0$

    将式子左边括号中转成一个数列a，得$a_n=a_{n-1}b+a_{n-2}\frac{d-b^2}{4}$

    这个数列可通过矩阵快速幂求得，而减号后的只会为0或者1（按照数据范围讨论），特判即可。

    复杂度$O(\log^2n)$：因为这个模数很恶心，所以要再花一个log时间来快速乘法，使得不会爆longlong。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define P 7528443412579576937LL
using namespace std;
struct matrix{
    long long a[3][3];
    int clear(){memset(a,0,sizeof(a));}
}E;
long long add(long long x,long long y){
    long long ans=-P+x+y;
    if(ans<0)ans+=P;
    return ans;
}
long long mul(long long x,long long y){
    long long ans=0;
    while(y){
        if(y&1)ans=add(ans,x);
        x=add(x,x);
        y>>=1;
    }
    return ans;
}
matrix operator * (matrix a,matrix b){
    matrix ans;
    ans.clear();
    for(int i=1;i<=2;i++){
        for(int j=1;j<=2;j++){
            for(int k=1;k<=2;k++){
                ans.a[i][j]=add(mul(a.a[i][k],b.a[k][j]),ans.a[i][j]);
            }
        }
    }
    return ans;
}
matrix power(matrix x,long long k){
    matrix ans;
    ans.clear();
    ans=E;
    while(k){
        if(k&1)ans=ans*x;
        x=x*x;
        k>>=1; 
    }
    return ans;
}
int main(){
    E.a[1][1]=1;
    E.a[2][2]=1;
    long long b,d,n;
    scanf("%lld%lld%lld",&b,&d,&n);
    matrix o;
    o.clear();
    o.a[1][1]=2;
    o.a[1][2]=b;
    matrix p;
    p.clear();
    p.a[2][1]=1;
    p.a[2][2]=b;
    p.a[1][2]=(d-b*b)/4;
    o=o*power(p,n);
    printf("%lld",o.a[1][1]-(b*b!=d&&n%2==0));
}

NOI2014 购票

2018-07-05
#NOI #斜率优化 #CDQ分治 #点分治

题目大意：

    有一棵树，对于每个点v有四个参数,$s_v$,$p_v$,$q_v$,$l_v$。

    $s_v$表示当前点到父亲节点的距离。如果d为每次跳跃的距离，那么每次跳跃的费用为$dp_v+q_v$。现在所有点都要去到1号点，并且他们每次往上跳不能超过距离$l_v$。对于每个点求最小花费。

    $n\leq2*10^5$,其余所有运算不超过longlong。

题解：

    链上做法：斜率优化

    设$d_i$为1号点到i号点的距离

    $f_i=min(f_j+(d_i-d_j)p_i+q_i)$

    设选k由于j且$d_k>d_j$：

    $f_k+(d_i-d_k)p_i+q_i<f_j+(d_i-d_j)p_i+q_i$

    整理得：

    $\frac{f_k-f_j}{d_k-d_j}<p_i$

    树链剖分+凸包合并$O(n\log^3 n)$：

    对于一条链来说，直接斜率优化即可，那么对于树上，我们可以按照dfs的顺序，一次加入点，并将树树剖，依次求出f值，在插入线段树的时候，只有左右儿子都是满时才合并，因为没加入的都不会被查询，所以可在$(n\log^3n)$的时间里求出答案（树剖一条链会有log段，限制$l_i$在线段树上取区间总共有log个，每个区间二分得到答案又一个log）（因为树剖常数小所以能过qwq）（然而懒惰的我写了下面的方法。）

    CDQ+树分治+凸包合并$O(n\log^2n)$：

    对于每次分治，把重心堵住，不停地堵住，然后利用当前节点到重心的链去更新重心答案，再递归计算。（看代码应该极好理解）

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 420000
#define slope(x,y) ((f[y]-f[x]+0.0)/(d[y]-d[x]))
#define cal(i,j) (0LL+f[j]+(d[i]-d[j])*p[i]+q[i])
using namespace std;
long long n,fa[N],tot,h[N],v[N],Next[N],siz[N],g[N],cnt,Q[N],o[N];
long long p[N],q[N],L[N],d[N],w[N],f[N];
struct node{long long w,id;}a[N];
bool cmp(node x,node y){return x.w>y.w;}
int add(int x,int y,long long z){
    Next[++tot]=h[x];
    h[x]=tot;
    v[tot]=y;
    w[tot]=z;
}
int dfs(int x){
    siz[x]=1;
    for(int i=h[x];i;i=Next[i]){
        d[v[i]]=d[x]+w[i];
        dfs(v[i]);
        siz[x]+=siz[v[i]];
    }
}
int get(int x,int S,int &rt){ 
    g[x]=0; siz[x]=1;
    for(int i=h[x];i;i=Next[i]){
        if(o[v[i]])continue;
        get(v[i],S,rt);
        siz[x]+=siz[v[i]];
        g[x]=max(g[x],siz[v[i]]);
    }
    g[x]=max(g[x],S-siz[x]);
    if(g[x]<g[rt]&&siz[x]>1)rt=x;
}
int dfs2(int x){
    a[++cnt].id=x;
    a[cnt].w=d[x]-L[x];
    for(int i=h[x];i;i=Next[i])if(!o[v[i]])dfs2(v[i]);
}
int solve(int x,int S){
    if(x==2){
        x++;
        x--;
    }
    if(S==1)return 0;
    int rt=0,now;
    get(x,S,rt);
    for(int i=h[rt];i;i=Next[i])o[v[i]]=1;
    solve(x,S-siz[rt]+1);
    cnt=0;
    for(int i=h[rt];i;i=Next[i])dfs2(v[i]);
    sort(a+1,a+cnt+1,cmp);
    now=rt;
    int top,l,r,mid,pos;
    top=0;
    for(int i=1;i<=cnt;i++){
        while(now!=fa[x]&&d[a[i].id]-L[a[i].id]<=d[now]){
            while(top>1&&slope(Q[top],now)>=slope(Q[top-1],Q[top]))top--;
            Q[++top]=now;now=fa[now];
        }
        if(top>0){
            l=1;r=top;pos=1;
            while(l<=r){
                mid=(l+r)>>1;
                if(mid==top){pos=top;break;}
                if(slope(Q[mid],Q[mid+1])>=p[a[i].id])l=mid+1,pos=mid+1;
                else r=mid-1;
            }
            f[a[i].id]=min(f[a[i].id],cal(a[i].id,Q[pos]));
        }
    }
    for(int i=h[rt];i;i=Next[i])solve(v[i],siz[v[i]]);
}
int main(){
    long long x;
    scanf("%lld%lld",&n,&x);
    for(int i=2;i<=n;i++){
        scanf("%lld%lld%lld%lld%lld",&fa[i],&x,&p[i],&q[i],&L[i]);
        add(fa[i],i,x);
    }
    dfs(1);
    g[0]=n+1;
    for(int i=2;i<=n;i++)f[i]=1LL<<60;
    solve(1,siz[1]);
    for(int i=2;i<=n;i++)printf("%lld\n",f[i]);
}

NOI2014 随机数生成器

2018-07-02
#NOI #卡常 #贪心 #模拟

题目大意：

    给出一个随机生成的矩阵（数字为1到n*m），求字典序最小的排序后的从$(1,1)$到$(n,m)$的数字序列。

题解：

    按照题意模拟出来之后从小到大贪心，每次加入点之后把不可能的情况排除，直接暴力排除（遇到被排除的就退），因为每个点最多只会被排除一次，所以排除的复杂度是$O(nm)$的，模拟出矩阵也是$O(nm)$的，总复杂度为$O(n*m)$。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define swap(x,y) t=x;x=y;y=t;
using namespace std;
int x0,a,b,c,d,n,m,q,tot;
int f[26000000],p[26000000],l[10200],r[10200];
static int t;
int main(){
    scanf("%d%d%d%d%d",&x0,&a,&b,&c,&d);
    scanf("%d%d%d",&n,&m,&q);
    for(int i=1;i<=n*m;i++)f[i]=i;
    for(int i=1;i<=n*m;i++){
        x0=(1LL*a*x0*x0+1LL*b*x0+c)%d;
        swap(f[i],f[x0%i+1]);
    }
    static int x,y;
    for(static int i=1;i<=q;i++){
        scanf("%d%d",&x,&y);
        swap(f[x],f[y]);
    }
    for(static int i=1;i<=n*m;i++)p[f[i]]=i;
    for(static int i=1;i<=n*m;i++){
        bool ooo=true;
        int y=(p[i]-1)%m+1;
        int x=(p[i]-y)/m+1;
        if(p[i]==0)continue;
        for(int i=x-1;i>=1;i--){
            for(int j=y+1;j<=m;j++){
                int ooo=((i-1)*m+j);
                if(p[f[ooo]]==0)break;
                p[f[ooo]]=0;
            }
        }
        for(int i=x+1;i<=n;i++){
            for(int j=y-1;j>=1;j--){
                int ooo=((i-1)*m+j);
                if(p[f[ooo]]==0)break;
                p[f[ooo]]=0;
            }
        }
        if(ooo==false)continue;
        l[++tot]=x;
        r[tot]=y;
        printf("%d ",i);
        if(tot==n+m-1)return 0; 
    }
}

NOI2014 魔法森林

2018-07-02
#NOI #SPFA

题目大意：

    一个无向图，每条边两个值a和b，求1到n路径上的$min(max(a_i)+max(b_j))$。

题解：

    按照a为第一关键字，b为第二关键字排序，按顺序加边，再松弛，因为每条边松弛一次，所以跟整张图的SPFA是一个复杂度。相当于对于每个a的可能值都做一次b的最短路，约等于优化暴力（spaly维护最小生成树的做法我不会啊qwq）

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 12000000
using namespace std;
struct node{
    int x,y,a,b;
}a[N];
int tot,Next[N],v[N],w[N],h[N],d[N]; 
int q[N],p[N],n,m;
int add(int x,int y,int z){
    tot++;
    Next[tot]=h[x];
    v[tot]=y;
    w[tot]=z;
    h[x]=tot;
}
bool cmp(node x,node y){
    if(x.a==y.a)return x.b<y.b;
    return x.a<y.a;
}
int spfa(int x){
    q[1]=x;
    if(x==1){
        x++;
        x--;
    }
    int head=0,tail=1;
    p[x]=1;
    while(head<tail){
        int now=q[++head];
        if(d[now]==0)continue;
        for(int i=h[now];i;i=Next[i]){
            if(max(d[now],w[i])<d[v[i]]||d[v[i]]==0){
                d[v[i]]=max(d[now],w[i]);
                if(p[v[i]]==0){
                    p[v[i]]=1;
                    q[++tail]=v[i];
                }
            }
        }
        p[now]=0;
    }
    p[x]=0;
}
int main(){
    d[1]=1;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++)scanf("%d%d%d%d",&a[i].x,&a[i].y,&a[i].a,&a[i].b);
    sort(a+1,a+1+m,cmp);
    int ans=1000000000,bns=1000000000;
    for(int i=1;i<=m;i++){
        add(a[i].x,a[i].y,a[i].b);
        add(a[i].y,a[i].x,a[i].b);
        spfa(a[i].x);
        spfa(a[i].y);
        if(d[n]==0)continue;
        bns=a[i].a+d[n];
        if(bns<ans)ans=bns;
    }
    spfa(1);
    if(ans==1000000000)ans=-1;
    printf("%d",ans);
}

NOI2014 动物园

#倍增 #NOI #KMP #卡常

题目大意：

    对于S的每个前缀T，对于T求出有多少个长度小于|T|的前缀和后缀相同，讲每个前缀T的答案加一乘起来膜$1000000007$。

题解：

    直接求Next数组然后倍增就好了。（倍增前后坐标相反能快差不多一倍，不然卡不过去qwq）

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 1200000
#define P 1000000007
using namespace std;
char s[N];
long long f[21][N],ans;
int Next[N],q[N],g[N];
int cal(int x){
    if(f[0][x]==0)return 0;
    int o=x;
    for(int i=20;i>=0;i--)
        if(f[i][o]>x/2)o=f[i][o];
    return f[0][o];
}
int get(){
    memset(f,0,sizeof(f));
    memset(q,0,sizeof(q));
    memset(g,0,sizeof(g));
    memset(Next,0,sizeof(Next));
    int l=strlen(s+1);
    q[1]=1;
    for(int i=2;i<=l;i++){
        int j=Next[i-1];
        while(j&&s[j+1]!=s[i])j=Next[j];
        Next[i]=j+(s[j+1]==s[i]);
        q[i]=q[Next[i]]+1;
        f[0][i]=Next[i];
        for(int j=1;j<=20;j++){
            f[j][i]=f[j-1][f[j-1][i]];
            if(f[j][i]==0)break;
        }
    }
    ans=1;
    for(int i=1;i<=l;i++)g[i]=q[cal(i)];
    for(int i=1;i<=l;i++)(ans*=1LL*g[i]+1)%=P; 
    printf("%lld\n",ans);
}
int main(){
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%s",s+1);
        get();
    }
}

NOI2014 起床困难综合征

2018-07-02
#NOI #贪心

题目大意：

    给定一系列位运算操作，在1到m里选一个数，使这个数在经历一系列操作后值最大。

题解：

    对于每一位分别求出输入时是0和1为多少，从高位往低位贪心选择较大值，如果在相同满足要求的结果下选择输入较小的。

#include<iostream>
#include<stdio.h>
#include<string.h> 
using namespace std;
long long n,m,d,ans,l;
char s[100];
int main(){
    scanf("%lld%lld",&n,&m);
    long long x=0,y=(1LL<<32)-1;
    for(int i=1;i<=n;i++){
        scanf("%s",s+1);
        scanf("%lld",&d);
        if(s[1]=='O')x=x|d,y=y|d;
        if(s[1]=='A')x=x&d,y=y&d;
        if(s[1]=='X')x=x^d,y=y^d;
    }
    for(int i=32;i>=0;i--){
        if(x&(1LL<<(i))){
            ans+=1LL<<i;
        }else{
            if(y&(1LL<<i)){
                if(l+(1LL<<i)<=m){
                    ans+=1LL<<i;
                    l+=1LL<<i;
                }
            }
        }
    }
    printf("%lld",ans);
}

NOI2007 生成树计数

2018-06-30
#NOI #状压DP #最小表示法

题目大意：

    有一条链，链上长度不超过k的点之间都会进行连边，求生成树个数。

题解：

    60%做法：$O(n^3)$,$n \leq 100$

    那么直接连边按照，用矩阵树定理高斯消元解行列式即可。

 因为$n\leq100$那么直接连边按照，用矩阵树定理$n^3$高斯消元解行列式即可。

    100%做法：$O(l^3log(n))$,$n\leq 10^{15}$,l为所有联通形式种类，当k=5时l=52。

    考虑到对于第i-k号点，如果最后会成为生成树，他必定与i-k+1到i号点中至少一个是一个联通快的。因此，我们维护$f_{i,j}$表示，在当前第i个点，前j个点的联通块是怎么样的，每次转移则是用第i号点与前面的连边情况进行转移，维护判断i-k号点在转移后是否与i-k+1到i号点中的部分点联通。搜索我们可以得到转移矩阵。

    初始条件：因为在前k个点中，所有点之间都是两两相连的，n个点的完全图的最小生成树是$n^{n-2}$，这个可用purfer序列简易证明。

    关于最小表示法：既联通情况2,2,3,1与联通情况1,1,2,3是相同的，所有本质相同的情况我们用它的最小表示来识别它。

#include<iostream>
#include<stdio.h>
#include<string.h>

#define N 120
#define P 65521
using namespace std;
int cnt,k,tot,l;
int a[10],b[10],q[60];
long long n;
map<int,int>p;
struct matrix{
    long long a[N][N];
    int clear(){memset(a,0,sizeof(a));}
    int E(){for(int i=1;i<=cnt;i++)a[i][i]=1;}
}g,f;
matrix operator *(matrix a,matrix b){
    matrix ans;
    ans.clear();
    for(int i=1;i<=cnt;i++){
        for(int j=1;j<=cnt;j++){
            for(int k=1;k<=cnt;k++){
                (ans.a[i][j]+=a.a[i][k]*b.a[k][j])%=P;
            }
        }
    }
    return ans;
}
matrix power(matrix x,long long k){
    matrix ans;
    ans.clear();
    ans.E();
    while(k){
        if(k&1)ans=ans*x;
        x=x*x;
        k>>=1;
    }
    return ans;
}
int get(int x){
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    int tot=0;
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3);
    for(int i=1;i<=k;i++){
        if(b[a[i]])a[i]=b[a[i]];
        else b[a[i]]=++tot,a[i]=tot;
    }
    int y=0;
    for(int i=1;i<=k;i++)y=y<<3,y+=a[i];
    return y;
}
int powerr(int x,int k){
    int ans=1;
    if(k<=0)return ans;
    while(k){
        if(k&1)(ans*=x)%=P;
        (x*=x)%=P;
        k>>=1;
    }
    return ans;
}
int gett(int x){
    int ans=1;
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3),b[a[i]]++;
    for(int i=1;i<=k;i++)ans*=powerr(b[i],b[i]-2);
    return ans;
}

    if(x==l){
        int y=get(o);

NOI2007 项链工厂

2018-06-28
#NOI #线段树

题目大意：

   维护一个环，和一个指针，有以下几种操作（该指针的下标为1，其余为位置的下标顺时针递增）:

   1:将指针逆时针移动k位

   2:将环以指针做对称轴翻转

   3:交换l位置和r位置上的颜色

   4:将l到r染成c颜色

   5:查询当前项链有多少段颜色（首尾相同算一段）

   6: 查询l到r有多少段颜色

题解：

   线段树

   对于翻转操作，我们修改对应下标，然后更新即可，需要完成的模块有区间更新（lazy标记），区间查询，单点更新，单点查询。

   （由于代码是在拍宣传片时候写的，思维极度混乱，由于实现问题甚至需要开八倍空间，这份代码极度丑陋qwq）。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 2200000
using namespace std;
int a[N],n,c,ooo;
struct node{
    int lx[N],rx[N],tag[N],t[N],ll,rr,ans;
    int pushdown(int i){
        if(tag[i]){
            lx[i]=tag[i];
            rx[i]=tag[i];
            t[i]=1;
            tag[i*2]=tag[i];
            tag[i*2+1]=tag[i];
            tag[i]=0;
            return 0;
        }
    }
    int merge(int i,int l,int r){
        if(tag[l])pushdown(l);
        if(tag[r])pushdown(r);
        t[i]=t[l]+t[r]-(rx[l]==lx[r]);
        lx[i]=lx[l];
        rx[i]=rx[r];
    }
    int build(int i,int l,int r){
        if(l==r){
            lx[i]=a[l];
            rx[i]=a[l];
            t[i]=1;
            return 0;
        }
        int mid=(l+r)/2;
        build(i*2,l,mid);
        build(i*2+1,mid+1,r);
        merge(i,i*2,i*2+1);
    }
    int insert(int i,int l,int r,int x,int y,int c){
        pushdown(i);
        if(x<=l&&r<=y){
            tag[i]=c;
            pushdown(i);
            merge(i,i*2,i*2+1);
            return 0;
        }
        int mid=(l+r)/2;
        if(mid>=x)insert(i*2,l,mid,x,y,c);
        if(mid<y)insert(i*2+1,mid+1,r,x,y,c);
        merge(i,i*2,i*2+1);
    }
    int add(int i){
        if(ll==0&&rr==0){
            ll=lx[i],rr=rx[i];
            ans=t[i];
            return 0;
        }
        ans=t[i]+ans-(rr==lx[i]);
        rr=rx[i];
        return 0;
    }
    int query(int i,int l,int r,int x,int y){
        pushdown(i); 
        if(x<=l&&y>=r){
            add(i);return ans;
        }
        int mid=(l+r)/2;
        if(x<=mid)query(i*2,l,mid,x,y);
        if(y>mid)query(i*2+1,mid+1,r,x,y);
        merge(i,i*2,i*2+1);
        return ans;
    }
    int clear(){
        ll=0,rr=0;
        ans=0;
        return 0;
    }
    int insert(int i,int l,int r,int x,int a){
        pushdown(i); 
        if(l==r){
            t[i]=1;
            lx[i]=a;
            rx[i]=a;
            return 0;
        }
        int mid=(l+r)/2;
        if(x<=mid)insert(i*2,l,mid,x,a);
        else insert(i*2+1,mid+1,r,x,a);
        merge(i,i*2,i*2+1);
    }
    int query_color(int i,int l,int r,int x){
        pushdown(i);
        if(l==r)return lx[i];
        int mid=(l+r)/2;
        if(x<=mid)return query_color(i*2,l,mid,x);
        else return query_color(i*2+1,mid+1,r,x);
        merge(i,i*2,i*2+1);
    }
}s;
int q;
char ss[10];
int get(int &x){
    x%=n;x+=n;x%=n;
    if(x==0)x=n;
}
int tot;
int main(){
    int p=0;
    scanf("%d%d",&n,&c);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    s.build(1,1,n);
    scanf("%d",&q);
    int o=1;
    for(int i=1;i<=q;i++){
        scanf("%s",ss+1);
        if(ss[1]=='R'){
            int k;
            scanf("%d",&k);
            if(p==0)o-=k;
            else o+=k;
            get(o);
        }
        if(ss[1]=='F'){p^=1;}
        if(ss[1]=='S'){
            int l,r;
            scanf("%d%d",&l,&r);
            if(p==0)l+=o-1,r+=o-1;
                else{
                    l=o+1-l,r=o+1-r;
                }
            get(l),get(r);
            int lx=s.query_color(1,1,n,l),rx=s.query_color(1,1,n,r);
            s.insert(1,1,n,r,lx);
            s.insert(1,1,n,l,rx); 
        }
        if(ss[1]=='P'){
            int l,r,x;
            scanf("%d%d%d",&l,&r,&x);
            if(p==0)l+=o-1,r+=o-1;
                else{
                    l=o+1-l,r=o+1-r;
                    swap(l,r);
                }
            get(l),get(r);
            if(l<=r)s.insert(1,1,n,l,r,x);
            else s.insert(1,1,n,1,r,x),s.insert(1,1,n,l,n,x);
        } 
        if(ss[1]=='C'){
            tot++;
//          if(tot==11){
//              printf("\n");
//              for(int i=1;i<=n;i++)printf("%d ",s.query_color(1,1,n,i));
//              printf("\n");
//          }
            int l=1,r=n,ooo=1;
            if(ss[2]=='S'){
                scanf("%d%d",&l,&r);
                if(p==0)l+=o-1,r+=o-1;
                else{
                    l=o+1-l,r=o+1-r;
                    swap(l,r);
                }
                get(l);get(r);
            }
            int ans=0;
            s.clear();
            if(l<=r)ans=s.query(1,1,n,l,r)-(r-l+1==n?(s.query_color(1,1,n,1)==s.query_color(1,1,n,n)):0);
            else{

                ans+=s.query(1,1,n,l,n)-(s.query_color(1,1,n,1)==s.query_color(1,1,n,n));
            }
            printf("%d\n",max(ans,1));
        }
    }

NOI2007 货币兑换

2018-06-28
#NOI #斜率优化 #CDQ分治

题目大意：

    股市有n天，有A和B两只股票，给出$a_i$，$b_i$表示A劵和B劵的价值，$r_i$表示每天提供的A劵与B劵的比例（必须按照这个比例来买）。

    给出开局的金额S，求n天后最多能得到的钱。

题解：

    易证每天必须全部卖出或者全部买入

    60%做法，$O(n^2)dp$：

    $f_i=\max(f_{i-1},A_ja_i+B_jb_i)$

    100%做法，$O(nlog^2n)$斜率优化+cdq分治:

    若k优于j且$A_k$>$A_j$，则：

    $A_ja_i+B_jb_i<A_ka_i+B_kb_i$

    $b_i(B_j-B_k)<a_i(A_k-A_j)$

    $-\frac{A_k-A_j}{B_k-B_j}<\frac{a_i}{b_i}$

    $\frac{A_k-A_j}{B_k-B_j}>-\frac{a_i}{b_i}$

    考虑将$B_i$作为横坐标$A_i$作为纵坐标，对于每个i来说，取最优答案即在上凸包做斜率为$-\frac{a_i}{b_i}$切线处取点。

    CDQ分治：solve(l,r)

    先按斜率$-\frac{a_i}{b_i}$排序。在solve(l,r)把时间小于等于mid的放到数组前半段，大于mid的放到数组后半段，solve(l,mid)，并将其暴力构建凸包，对mid+1到r的答案进行更新（这时满足斜率单调的条件，而最先开始的按照斜率排序保证了后半段的查询中斜率式单调递增的）。然后solve(mid+1,r)，最后将l到r中的点按照横纵坐标排序（为了返回之后能构建凸包）。

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<cmath>
#define eps 1e-9
#define N 1200000
using namespace std;
struct node{
    double a,b,k,x,y,r;
    int id;
}a[N],t[N];
int s[N];
double f[N];
int n;
bool cmp(node x,node y){
    return x.k>y.k;
}
bool cmp1(node x,node y){
    if(fabs(x.x-y.x)<eps)return x.y<y.y;
    return x.x<y.x;
}
bool slope(node x,node y,node z){
    return (z.x-x.x)*(y.y-x.y)-eps<(y.x-x.x)*(z.y-x.y);
}
double slope(node x,node y){
    if(fabs(x.x-y.x)<eps)return 1e10;
    return (y.y-x.y)/(y.x-x.x);
}
int solve(int l,int r){
    if(l==r){
        f[l]=max(f[l],f[l-1]);
        a[l].y=f[l]/(a[l].a*a[l].r+a[l].b);
        a[l].x=a[l].y*a[l].r;
        return 0;
    }
    int mid=(l+r)/2;
    int l1=l-1,l2=mid;
    for(int i=l;i<=r;i++)if(a[i].id<=mid)t[++l1]=a[i];else t[++l2]=a[i];
    for(int i=l;i<=r;i++)a[i]=t[i];
    solve(l,mid);
    int top=0;
    for(int i=l;i<=mid;i++){
        while(top>1&&slope(a[s[top-1]],a[s[top]],a[i]))top--;
        s[++top]=i;
    }
    int j=1;
    for(int i=mid+1;i<=r;i++){
        while(j<top&&slope(a[s[j]],a[s[j+1]])+eps>a[i].k)j++;
        int now=a[i].id,o=s[j];
        f[now]=max(f[now],a[i].a*a[o].x+a[i].b*a[o].y);
    }
    solve(mid+1,r);
    l1=l,l2=mid+1;
    for(int i=l;i<=r;i++)if((l2>r)||(l1<=mid&&cmp1(a[l1],a[l2])))t[i]=a[l1++];else t[i]=a[l2++];
    for(int i=l;i<=r;i++)a[i]=t[i];
}
int main(){ 
    scanf("%d%lf",&n,&f[0]);
    for(int i=1;i<=n;i++){
        scanf("%lf%lf%lf",&a[i].a,&a[i].b,&a[i].r);
        a[i].k=-a[i].a/a[i].b;
        a[i].id=i;
    }
    sort(a+1,a+1+n,cmp);
    solve(1,n);
    printf("%.5lf\n",f[n]);
}
